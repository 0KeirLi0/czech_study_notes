\section{Verb In Present Tense}

\subsection{Infinitive -AT (-Á- Type)}
To conjugate \textbf{-at} type verbs, simply remove the final \textbf{-t} and add the appropriate personal endings to the long-vowel stem \textbf{-á-}.
\begin{center}
\begin{tabular}{c|c|c|c}
\hline
    \textbf{Person} & \textbf{Ending} & \textbf{Dělat (To do)} & \textbf{Mít (To have)} \\ \hline
    Já & \textbf{-m} & dělá\textbf{m} & má\textbf{m} \\
    Ty & \textbf{-š} & dělá\textbf{š} & má\textbf{š} \\
    On/Ona/To & \textbf{---} & dělá & má \\ \hline
    My & \textbf{-me} & dělá\textbf{me} & má\textbf{me} \\
    Vy & \textbf{-te} & dělá\textbf{te} & má\textbf{te} \\
    Oni & \textbf{-jí} & dělá\textbf{jí} & ma\textbf{jí} \\ \hline
\end{tabular}
\end{center}

\subsection{Infinitive -IT / -ET / -ĚT (-Í- Type)}
For verbs ending in \textbf{-it}, \textbf{-et}, or \textbf{-ět}, the stem vowel changes to or remains a long \textbf{-í-} before adding the personal endings.
\begin{center}
\begin{tabular}{c|c|c|c}
\hline
    \textbf{Person} & \textbf{Ending} & \textbf{Mluvit (To speak)} & \textbf{Rozumět (To understand)} \\ \hline
    Já & \textbf{-m} & mluví\textbf{m} & rozumí\textbf{m} \\
    Ty & \textbf{-š} & mluví\textbf{š} & rozumí\textbf{š} \\
    On/Ona/To & \textbf{---} & mluví & rozumí \\ \hline
    My & \textbf{-me} & mluví\textbf{me} & rozumí\textbf{me} \\
    Vy & \textbf{-te} & mluví\textbf{te} & rozumí\textbf{te} \\
    Oni & \textbf{-í} & mluví & rozumí \\ \hline
\end{tabular}
\end{center}

\subsection{Infinitive -OVAT (-UJ- Type)}
For verbs ending in \textbf{-ovat}, replace the suffix with \textbf{-uj-} before adding the personal endings.
\begin{center}
\begin{tabular}{c|c|c|c}
\hline
    \textbf{Person} & \textbf{Ending} & \textbf{Pracovat (To work)} & \textbf{Studovat (To study)} \\ \hline
    Já & \textbf{-u/i*} & pracuj\textbf{u} & studuj\textbf{u} \\
    Ty & \textbf{-eš} & pracuj\textbf{eš} & studuj\textbf{eš} \\
    On/Ona/To & \textbf{-e} & pracuj\textbf{e} & studuj\textbf{e} \\ \hline
    My & \textbf{-eme} & pracuj\textbf{eme} & studuj\textbf{eme} \\
    Vy & \textbf{-ete} & pracuj\textbf{ete} & studuj\textbf{ete} \\
    Oni & \textbf{-ou/í*} & pracuj\textbf{ou} & studuj\textbf{ou} \\ \hline
\end{tabular}
\end{center}
*endings \textbf{-i/-í} are more formal, used in formal or written Czech.
\subsection{Infinitive -T (-U- Type)}
These verbs often have short stems. The thematic vowel is \textbf{-u-}, followed by the standard consonant endings.
\begin{center}
\begin{tabular}{c|c|c|c}
\hline
    \textbf{Person} & \textbf{Ending} & \textbf{Pít (To drink)} & \textbf{Psát (To write)} \\ \hline
    Já & \textbf{-u} & pij\textbf{u} & píš\textbf{u} \\
    Ty & \textbf{-eš} & pij\textbf{eš} & píš\textbf{eš} \\
    On/Ona/To & \textbf{-e} & pij\textbf{e} & píš\textbf{e} \\ \hline
    My & \textbf{-eme} & pij\textbf{eme} & píš\textbf{eme} \\
    Vy & \textbf{-ete} & pij\textbf{ete} & píš\textbf{ete} \\
    Oni & \textbf{-ou} & pij\textbf{ou} & píš\textbf{ou} \\ \hline
\end{tabular}
\end{center}
